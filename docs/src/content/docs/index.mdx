  <div class="dux-section-head">
    <span class="eyebrow">In one screen</span>
    <h2>Define, sync, query</h2>
    <p>The whole loop — model, schema sync, typed CRUD — without leaving Go or running a tool.</p>
  </div>

```go
type Product struct {
    oro.Model
    Code  string
    Price uint
}

func (Product) Define(s *oro.SchemaBuilder) {
    s.Table("products")
    s.Field("Code").String().Unique()
    s.Field("Price").Uint().Default(0)
}

// register + sync the schema from the definition above
db.Register(Product{})
db.Sync(ctx)

// typed, readable CRUD with semantic conditions
product, _ := db.Use[Product]().Create(ctx, &Product{Code: "P001", Price: 100})
found, _ := db.Use[Product]().Where(oro.Field("Code").Eq("P001")).First(ctx)
```
